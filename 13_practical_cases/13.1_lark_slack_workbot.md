## 13.1 实战案例1：企业飞书/Slack群工作助手（从零到部署）

本节通过一个完整的企业场景案例，演示如何从零开始构建一个多渠道工作助手，包括需求分析、架构设计、配置实现、核心代码、测试验证和部署上线的完整闭环。

### 13.1.1 需求分析与场景设定

#### 背景故事

假设你在一家中等规模的互联网公司（500人左右）担任技术负责人。公司研发团队日常工作分散在飞书（内部）和 Slack（跨国团队）两个平台。现在需要部署一个 **智能工作助手**，帮助团队自动化以下日常工作：

1. **日报汇总**：每天16:00 自动从研发群提取 @助手 的进度更新，聚合成结构化日报发至指定文档。
2. **代码评审辅助**：当工程师在群里 @ 助手并贴上 GitHub PR 链接时，能自动拉取代码、总结改动、识别风险点，生成评审意见。
3. **故障诊断**：接收故障告警，查询内部知识库和监控系统，协助定位根因并建议快速修复方案。
4. **工时追踪**：从群聊消息中自动识别完成的任务，更新项目管理系统（Jira/飞书任务）。

#### 技术约束与目标

| 维度 | 需求 | 实现影响 |
|-----|------|--------|
| **可用性** | 支持飞书和 Slack 同步工作 | 需要多渠道路由与消息映射 |
| **权限** | 不同团队有不同权限（研发可查源代码，HR不行） | 需要实现基于用户角色的访问控制 |
| **一致性** | 跨平台操作保证一致 | 需要统一的任务状态机与补偿机制 |
| **延迟** | 日报聚合应在16:05分前完成 | 需要预加载与缓存策略 |
| **审计** | 所有操作必须有日志与确认 | 需要完整的执行堆栈与人工审核流 |

```javascript
/**
 * 消息接收Hook：在消息进入Agent前的前置处理
 * 职责：权限检查、消息分类、缓存策略判断
 */

const CLEARANCE_LEVELS = {
  'engineer': 3,
  'manager': 2,
  'guest': 1
};

const TRIGGER_KEYWORDS = {
  'codereview': { tool: 'github_pr_analyzer', priority: 'high' },
  '故障': { tool: 'monitoring_query', priority: 'urgent' },
  '日报': { tool: 'document_writer', priority: 'normal' }
};

module.exports = {
  name: 'on_message_received',
  version: '1.0.0',

  async execute(context) {
    const { message, channel, sender, logger } = context;

    logger.info(`[Hook] Received message from ${sender.userId} in ${channel}`, {
      messageId: message.id,
      content: message.text.substring(0, 100)
    });

    // 1. 权限检查
    const userRole = await getUserRole(sender.userId);
    const clearanceLevel = CLEARANCE_LEVELS[userRole] || 1;

    if (clearanceLevel < 1) {
      logger.warn(`[Hook] User ${sender.userId} has insufficient clearance`, {
        requiredLevel: 1,
        actualLevel: clearanceLevel
      });
      return {
        allow: false,
        reason: 'insufficient_clearance',
        response: '抱歉，您没有权限使用此功能'
      };
    }

    // 2. 消息分类与优先级判断
    let triggerType = null;
    let priority = 'normal';
    for (const [keyword, config] of Object.entries(TRIGGER_KEYWORDS)) {
      if (message.text.includes(keyword)) {
        triggerType = config.tool;
        priority = config.priority;
        break;
      }
    }

    // 3. 缓存决策
    const shouldCache = {
      'github_pr_analyzer': true,
      'knowledge_base_search': true,
      'monitoring_query': false // 监控数据实时性要求高，不缓存
    }[triggerType] || false;

    logger.info(`[Hook] Message classification`, {
      triggerType,
      priority,
      shouldCache
    });

    return {
      allow: true,
      metadata: {
        userRole,
        clearanceLevel,
        triggerType,
        priority,
        shouldCache,
        receivedAt: Date.now()
      }
    };
  }
};

async function getUserRole(userId) {
  // 从内部目录服务查询用户角色
  // 实际场景可调用企业LDAP或组织管理系统
  const roleMap = {
    'user123': 'engineer',
    'manager456': 'manager',
    'guest789': 'guest'
  };
  return roleMap[userId] || 'guest';
}

文件路径：`hooks/before_tool_execution.js`

```javascript
/**
 * 工具执行前Hook：检查工具权限、超时、速率限制
 */

module.exports = {
  name: 'before_tool_execution',
  version: '1.0.0',

  async execute(context) {
    const { tool, args, message, metadata, logger } = context;
    const { clearanceLevel, userRole } = metadata;

    logger.info(`[Hook] Before tool execution`, {
      tool: tool.name,
      user: userRole,
      clearanceLevel
    });

    // 1. 权限检查
    const toolPolicy = await getToolPolicy(tool.name);
    if (toolPolicy.minClearance && clearanceLevel < toolPolicy.minClearance) {
      logger.error(`[Hook] Tool execution denied: insufficient clearance`, {
        tool: tool.name,
        required: toolPolicy.minClearance,
        actual: clearanceLevel
      });
      return {
        allow: false,
        reason: 'insufficient_clearance',
        errorMessage: `执行 ${tool.name} 需要权限等级 ${toolPolicy.minClearance}，您的等级是 ${clearanceLevel}`
      };
    }

    // 2. 参数安全检查（防止注入）
    const sanitizedArgs = sanitizeToolArgs(tool.name, args);

    // 3. 速率限制检查
    const rateLimitKey = `${userRole}:${tool.name}`;
    const isAllowed = await checkRateLimit(rateLimitKey);
    if (!isAllowed) {
      logger.warn(`[Hook] Rate limit exceeded`, { rateLimitKey });
      return {
        allow: false,
        reason: 'rate_limit_exceeded',
        errorMessage: '操作过于频繁，请稍候再试'
      };
    }

    // 4. 审计记录
    await logAuditEvent({
      action: 'TOOL_EXECUTION_REQUESTED',
      tool: tool.name,
      user: userRole,
      timestamp: Date.now(),
      args: JSON.stringify(sanitizedArgs),
      status: 'PENDING_APPROVAL'
    });

    return {
      allow: true,
      sanitizedArgs,
      timeout: toolPolicy.timeout || 30000
    };
  }
};

async function getToolPolicy(toolName) {
  // 返回工具策略（实际应从配置加载）
  const policies = {
    'github_pr_analyzer': { minClearance: 2, timeout: 30000 },
    'jira_issue_manager': { minClearance: 2, timeout: 20000 },
    'knowledge_base_search': { minClearance: 1, timeout: 15000 },
    'monitoring_query': { minClearance: 2, timeout: 20000 },
    'document_writer': { minClearance: 2, timeout: 25000 }
  };
  return policies[toolName] || {};
}

function sanitizeToolArgs(toolName, args) {
  // 根据工具类型清理参数
  if (toolName === 'github_pr_analyzer') {
    // 确保只能访问公开或授权的仓库
    if (args.repo) {
      args.repo = args.repo.replace(/[^a-zA-Z0-9-]/g, '');
    }
  }
  if (toolName === 'document_writer') {
    // 路径必须在允许的文件夹内
    const allowedPaths = ['/团队工作区/日报汇总', '/项目管理/进度'];
    if (!allowedPaths.some(p => args.path.startsWith(p))) {
      throw new Error(`Access denied to path: ${args.path}`);
    }
  }
  return args;
}

async function checkRateLimit(key) {
  // 使用Redis实现速率限制（伪代码）
  const redis = require('redis');
  const client = redis.createClient();
  const current = await client.incr(`ratelimit:${key}`);
  if (current === 1) {
    await client.expire(`ratelimit:${key}`, 60);
  }
  return current <= 10;
}

async function logAuditEvent(event) {
  // 写入审计日志
  const fs = require('fs').promises;
  await fs.appendFile(
    '/var/log/openclaw/audit.log',
    JSON.stringify(event) + '\n'
  );
}

#### 部署脚本

```bash
#!/bin/bash
# deploy/deploy.sh

set -e

NAMESPACE="ai-platform"
RELEASE_NAME="openclaw-work-assistant"

echo "=== OpenClaw 工作助手部署流程 ==="

# 1. 验证前置条件
echo "[1/6] 验证部署前置条件..."
kubectl cluster-info >/dev/null 2>&1 || {
  echo "ERROR: 无法连接到Kubernetes集群"
  exit 1
}

kubectl get namespace $NAMESPACE >/dev/null 2>&1 || {
  echo "创建命名空间: $NAMESPACE"
  kubectl create namespace $NAMESPACE
}

# 2. 创建密钥
echo "[2/6] 注入敏感信息到Secret..."
kubectl create secret generic openclaw-lark \
  --from-literal=app-id=$LARK_APP_ID \
  --from-literal=app-secret=$LARK_APP_SECRET \
  --namespace=$NAMESPACE \
  --dry-run=client -o yaml | kubectl apply -f -

kubectl create secret generic openclaw-slack \
  --from-literal=bot-token=$SLACK_BOT_TOKEN \
  --from-literal=app-token=$SLACK_APP_TOKEN \
  --namespace=$NAMESPACE \
  --dry-run=client -o yaml | kubectl apply -f -

# 3. 构建镜像
echo "[3/6] 构建Docker镜像..."
docker build -t openclaw:latest \
  -f Dockerfile \
  --build-arg VERSION=$(git describe --tags) \
  .

docker tag openclaw:latest $DOCKER_REGISTRY/openclaw:latest
docker push $DOCKER_REGISTRY/openclaw:latest

# 4. 部署应用
echo "[4/6] 部署到Kubernetes..."
kubectl apply -f deploy/k8s/openclaw-deployment.yaml

# 5. 等待就绪
echo "[5/6] 等待Pod就绪..."
kubectl rollout status deployment/openclaw-work-assistant \
  -n $NAMESPACE \
  --timeout=5m

# 6. 验证部署
echo "[6/6] 验证部署..."
GATEWAY_IP=$(kubectl get svc openclaw-gateway -n $NAMESPACE \
  -o jsonpath='{.status.loadBalancer.ingress[0].ip}')

if [ -z "$GATEWAY_IP" ]; then
  echo "WARNING: LoadBalancer IP未分配，使用NodePort"
  GATEWAY_PORT=$(kubectl get svc openclaw-gateway -n $NAMESPACE \
    -o jsonpath='{.spec.ports[0].nodePort}')
  GATEWAY_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  GATEWAY_URL="http://$GATEWAY_IP:$GATEWAY_PORT"
else
  GATEWAY_URL="http://$GATEWAY_IP:18789"
fi

echo ""
echo "✓ 部署成功!"
echo "Gateway URL: $GATEWAY_URL"
echo ""
echo "后续步骤:"
echo "1. 在飞书开放平台配置事件订阅，指向: $GATEWAY_URL/webhook/lark"
echo "2. 在Slack应用配置中设置事件URL: $GATEWAY_URL/webhook/slack"
echo "3. 验证连接: curl $GATEWAY_URL/health"
```bash

### 13.1.7 常见问题FAQ

**Q1：部署后飞书消息无法接收，怎么办？**

A：按以下顺序排查：

1. 确认Pod健康：`kubectl logs -f deploy/openclaw-work-assistant -n ai-platform`
2. 确认Gateway可访问：`curl http://<gateway-ip>:18789/health`
3. 确认飞书应用已发布：检查飞书开放平台应用状态
4. 确认事件订阅URL正确：在飞书平台检查“事件与回调”的URL是否与实际Gateway地址一致
5. 确认网络连接：尝试从飞书侧主动拉起连接测试

**Q2：如何监控OpenClaw的性能？**

A：使用以下方式：

```bash
# 查看实时日志
kubectl logs -f deploy/openclaw-work-assistant -n ai-platform --tail=100

# 查看Prometheus指标
kubectl port-forward svc/openclaw-gateway 9090:9090 -n ai-platform
# 访问 http://localhost:9090

# 查看审计日志
kubectl exec -it <pod-name> -n ai-platform -- tail -f /var/log/openclaw/audit.log
```text

**Q3：如何处理工具执行超时？**

A：在配置中调整超时值：

```json
{
  "tools": {
    "github_pr_analyzer": {
      "timeout": 45000,
      "maxAttempts": 3,
      "retryStrategy": "exponential_backoff"
    }
  }
}
```text

**Q4：如何安全地更新敏感信息（如Token）？**

A：不需要重新部署，直接更新Secret：

```bash
kubectl patch secret openclaw-lark \
  -p '{"data":{"app-secret":"'$(echo -n $NEW_SECRET | base64)'"}}'  \
  -n ai-platform
```text

OpenClaw会自动热加载Secret变更（无需重启）。
