#### 模型对比与应用场景

| 模型名称 | 上下文窗口 | 推理能力 | 速度 | 成本（输入/输出） |
|---------|-----------|---------|------|------|
| Claude Haiku 4.5 | 200K（可扩展至 1M） | ⭐⭐⭐ | 最快 | $1 / $5 per MTok |
| Claude Sonnet 4.6 | 200K（可扩展至 1M） | ⭐⭐⭐⭐ | 快 | $3 / $15 per MTok |

#### 模型选择决策树

```javascript
// 根据任务特征选择最优模型
class ModelSelector {
  selectModel(task) {
    const score = {
      complexity: this.estimateComplexity(task),      // 0-1
      contextSize: this.estimateContextSize(task),    // 0-1
      latencyRequirement: this.estimateLatency(task), // 0-1 (0=realtime, 1=batch)
      costSensitivity: this.estimateCost(task)        // 0-1 (0=must be cheap, 1=unlimited)
    };

    // 决策规则
    if (score.latencyRequirement > 0.8) {
      // 实时场景：优先选择快速模型
      return 'claude-haiku-4-5'; // 最快
    }

    if (score.complexity > 0.7 && score.contextSize > 0.7) {
      // 复杂+大上下文：使用最强模型
      return 'claude-opus-4-6'; // 最强
    }

    if (score.complexity > 0.6) {
      // 中等复杂：使用Sonnet
      return 'claude-sonnet-4-6'; // 最平衡
    }

    if (score.costSensitivity < 0.3) {
      // 成本敏感：使用最便宜的
      return 'claude-haiku-4-5'; // 最便宜
    }

    // 默认：使用Sonnet作为通用方案
    return 'claude-sonnet-4-6';
  }

  estimateComplexity(task) {
    // 基于任务特征评估复杂度
    let complexity = 0;
    if (task.requiresReasoning) complexity += 0.2;
    if (task.requiresPlanning) complexity += 0.2;
    if (task.requiresMathematicalCalculation) complexity += 0.2;
    if (task.requiresCodeGeneration) complexity += 0.2;
    if (task.requiresMultilingualSupport) complexity += 0.1;
    return Math.min(complexity, 1);
  }

  estimateContextSize(task) {
    // 英文约 4 字符 ≈ 1 token；中文约 1.5 字符 ≈ 1 token。
    // 通过检测 CJK 字符占比动态调整估算系数。
    const cjkRatio = this.estimateCJKRatio(task.documentContent || '');
    const charsPerToken = 4 * (1 - cjkRatio) + 1.5 * cjkRatio; // 加权平均
    const estimatedTokens = task.documentLength / charsPerToken +
                           task.conversationHistory.length * 50;
    return Math.min(estimatedTokens / 1000000, 1); // 当前模型支持最高 1M context window
  }

  estimateCJKRatio(text) {
    // 统计 CJK 统一表意文字占比（U+4E00–U+9FFF）
    const cjkChars = (text.match(/[\u4e00-\u9fff]/g) || []).length;
    return text.length > 0 ? cjkChars / text.length : 0;
  }

  estimateLatency(task) {
    // 0 = 需要实时（< 1s），1 = 异步批处理（可等待）
    if (task.isInteractive) return 0;
    if (task.isAsyncJob) return 1;
    return 0.5;
  }

  estimateCost(task) {
    // 0 = 成本关键，1 = 成本无关
    if (task.volumePerDay > 10000) return 0; // 高频任务要便宜
    if (task.isBackgroundJob) return 0.3;
    return 0.7; // 一般场景不太关心成本
  }
}

#### MCP服务器的生命周期管理

```javascript
class MCPServerManager {
  constructor() {
    this.servers = new Map();
    this.connections = new Map();
  }

  async startServer(serverId, config) {
    try {
      logger.info(`Starting MCP server: ${serverId}`);

      const server = this.createServer(config);
      await server.initialize();

      // 建立连接
      const connection = await server.connect({
        timeout: config.timeout || 30000,
        retryPolicy: {
          maxRetries: 3,
          backoffMs: 1000
        }
      });

      this.servers.set(serverId, server);
      this.connections.set(serverId, connection);

      // 健康检查
      await this.healthCheck(serverId);

      logger.info(`✓ MCP server started: ${serverId}`);
      return server;
    } catch (error) {
      logger.error(`Failed to start MCP server ${serverId}`, error);
      throw error;
    }
  }

  async stopServer(serverId) {
    const server = this.servers.get(serverId);
    if (server) {
      await server.disconnect();
      this.servers.delete(serverId);
      this.connections.delete(serverId);
      logger.info(`✓ MCP server stopped: ${serverId}`);
    }
  }

  async healthCheck(serverId) {
    const server = this.servers.get(serverId);
    if (!server) throw new Error(`Server not found: ${serverId}`);

    try {
      const health = await server.ping();
      if (!health.ok) {
        throw new Error(`Server health check failed: ${health.message}`);
      }
      logger.debug(`Health check passed: ${serverId}`);
    } catch (error) {
      logger.warn(`Health check failed for ${serverId}`, error);
      // 触发重启逻辑
      await this.restartServer(serverId);
    }
  }

  async restartServer(serverId) {
    const config = await this.getServerConfig(serverId);
    await this.stopServer(serverId);
    await new Promise(r => setTimeout(r, 1000)); // 等待1秒
    await this.startServer(serverId, config);
  }

  async executeToolViaServer(serverId, toolName, params) {
    const connection = this.connections.get(serverId);
    if (!connection) {
      throw new Error(`No active connection for server: ${serverId}`);
    }

    try {
      const result = await connection.executeTool(toolName, params, {
        timeout: 30000
      });
      return result;
    } catch (error) {
      logger.error(`Tool execution failed on ${serverId}:${toolName}`, error);
      throw error;
    }
  }

  createServer(config) {
    switch (config.type) {
      case 'stdio':
        return new StdioMCPServer(config);
      case 'http':
        return new HttpMCPServer(config);
      case 'built-in':
        return new BuiltInMCPServer(config);
      default:
        throw new Error(`Unknown server type: ${config.type}`);
    }
  }
}

```javascript
// 统一的搜索接口支持向量和全文混合搜索
class SearchIntegration {
  // 向量搜索
  async vectorSearch(query, topK = 5) {
    const embedding = await this.encoder.encode(query);
    const results = await this.vectorDB.search(embedding, {
      limit: topK,
      threshold: 0.5
    });
    return results;
  }

  // 全文搜索
  async fullTextSearch(query, filters = {}) {
    const results = await this.searchEngine.search(query, {
      filters,
      highlight: true
    });
    return results;
  }

  // 混合搜索：结合向量和关键词相关性
  async hybridSearch(query, topK = 5) {
    const [vectorResults, textResults] = await Promise.all([
      this.vectorSearch(query, topK * 2),
      this.fullTextSearch(query, {}, topK * 2)
    ]);

    // 合并结果，根据相关性排序
    const merged = this.mergeResults(vectorResults, textResults);
    return merged.slice(0, topK);
  }

  mergeResults(vectorResults, textResults) {
    const resultMap = new Map();

    // 添加向量搜索结果
    vectorResults.forEach((r, idx) => {
      resultMap.set(r.id, {
        ...r,
        vectorScore: 1 - (idx / vectorResults.length),
        textScore: 0
      });
    });

    // 添加全文搜索结果
    textResults.forEach((r, idx) => {
      const existing = resultMap.get(r.id);
      const textScore = 1 - (idx / textResults.length);

      if (existing) {
        existing.textScore = textScore;
      } else {
        resultMap.set(r.id, {
          ...r,
          vectorScore: 0,
          textScore
        });
      }
    });

    // 计算最终分数：加权平均
    const results = Array.from(resultMap.values()).map(r => ({
      ...r,
      score: r.vectorScore * 0.6 + r.textScore * 0.4
    }));

    return results.sort((a, b) => b.score - a.score);
  }
}

### 16.1.5 MCP最新功能与特性

#### MCP协议的最新发展（截至2026年3月）

MCP（Model Context Protocol）在2026年引入了三个重要新功能，进一步提升了Claude与外部系统的集成能力：

##### 1. Streamable HTTP传输

传统HTTP传输基于request-response模式。Streamable HTTP提供了服务器向客户端推送数据的能力，类似于Server-Sent Events（SSE）。

```python
class StreamableHTTPMCPServer:
    """支持Streamable HTTP的MCP服务器"""

    async def stream_data_resource(self, stream_id: str,
                                  resource_uri: str) -> AsyncGenerator:
        """
        通过流式传输发送资源数据

        优势：
        1. 实时数据流：对于日志、传感器数据等实时更新
        2. 减少延迟：不需要等待完整响应
        3. 降低内存：大文件可分块传输
        4. 双向通信：支持客户端反馈
        """

        try:
            # 建立流式连接
            connection = await self.establish_stream(resource_uri)

            # 持续推送数据
            async for chunk in connection.stream():
                yield {
                    "type": "resource_chunk",
                    "stream_id": stream_id,
                    "data": chunk,
                    "metadata": {
                        "chunk_size": len(chunk),
                        "timestamp": datetime.utcnow().isoformat()
                    }
                }

        except Exception as e:
            yield {
                "type": "stream_error",
                "stream_id": stream_id,
                "error": str(e)
            }

##### 2. Elicitation（双向提示）

Elicitation允许MCP服务器向Claude发送提示或查询，实现真正的双向通信。这突破了传统的“请求-响应”模式。

```python
class ElicitationMCPServer:
    """支持Elicitation的MCP服务器"""

    async def elicit_user_decision(self, context: Dict) -> Dict:
        """
        向Claude（进而向用户）发起提示

        使用场景：
        1. 数据歧义：需要用户澄清
        2. 决策支持：需要用户确认
        3. 补充信息：需要用户提供额外信息
        """

        elicitation_request = {
            "type": "elicitation",
            "query": "Multiple datasets found for 'sales_2024'. Which one should I use?",
            "options": [
                {
                    "id": "dataset_1",
                    "description": "Weekly sales aggregation",
                    "record_count": 52
                },
                {
                    "id": "dataset_2",
                    "description": "Daily sales detailed records",
                    "record_count": 365
                }
            ],
            "context": context,
            "timeout": 30  # 秒

        # 发送elicitation到Claude
        response = await self.send_elicitation(elicitation_request)

        # 获取用户选择

            "elicitation_id": response.get("id"),
            "selected_dataset": selected_dataset,
            "user_reasoning": response.get("reasoning", "")
        }

##### 3. OAuth 2.1授权

MCP 2.0引入了OAuth 2.1支持，提供了更安全的身份验证和授权机制。

```python
class MCPOAuth21Integration:
    """MCP OAuth 2.1集成"""

    async def setup_oauth_authentication(self, server_config: Dict) -> Dict:
        """
        配置OAuth 2.1认证

        支持的授权流程：
        1. Authorization Code Flow（推荐）
        2. PKCE（Public Clients）
        3. Client Credentials（服务间通信）

class MCPAuthenticationManager:
    """MCP认证管理器"""

    def __init__(self):
        self.oauth_clients = {}
        self.token_cache = {}

    async def authenticate_server(self, server_id: str,
                                 oauth_config: Dict) -> str:
        """
        使用OAuth 2.1认证MCP服务器
        """

        # 步骤1：生成授权请求
        auth_request = self._generate_auth_request(oauth_config)

        # 步骤2：获取用户授权
        auth_code = await self._get_user_authorization(auth_request)

        # 步骤3：交换访问令牌
        access_token = await self._exchange_authorization_code(
            auth_code,
            oauth_config
        )

        # 步骤4：缓存令牌
        self.token_cache[server_id] = {
            "access_token": access_token,
            "expires_at": self._calculate_expiry(),
            "scopes": oauth_config["scopes"]
        }

        return access_token

    async def _refresh_token_if_needed(self, server_id: str,
                                      oauth_config: Dict) -> str:
        """
        检查并自动刷新令牌

        OAuth 2.1推荐：实施Refresh Token Rotation
        """

        cached_token = self.token_cache.get(server_id)

        if not cached_token:
            return await self.authenticate_server(server_id, oauth_config)

        # 检查是否需要刷新
        if self._is_token_near_expiry(cached_token):
            new_token = await self._do_token_refresh(
                server_id,
                oauth_config
            )

            self.token_cache[server_id] = {
                "access_token": new_token,
                "expires_at": self._calculate_expiry(),
                "scopes": oauth_config["scopes"]
            }

            return new_token

        return cached_token["access_token"]
